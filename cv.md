function reverseWords(str) {
  const splitString = str.split(' ');
  let sumString = '';

  for(let i = 0; i <= splitString.length - 1; i++) {
    if(i == splitString.length - 1) {
      sumString += splitString[i].split('').reverse().join('');
    } else {
      sumString += splitString[i].split('').reverse().join('') + ' ';
    }
  }
  return sumString;
}
